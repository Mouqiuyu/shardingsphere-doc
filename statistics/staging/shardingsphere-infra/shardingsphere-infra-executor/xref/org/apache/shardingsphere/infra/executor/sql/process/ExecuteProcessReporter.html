18  package org.apache.shardingsphere.infra.executor.sql.process;
19  
20  import org.apache.shardingsphere.infra.binder.QueryContext;
21  import org.apache.shardingsphere.infra.executor.kernel.model.ExecutionGroupContext;
22  import org.apache.shardingsphere.infra.executor.sql.execute.engine.SQLExecutionUnit;
23  import org.apache.shardingsphere.infra.executor.sql.process.model.ExecuteProcessContext;
24  import org.apache.shardingsphere.infra.executor.sql.process.model.ExecuteProcessStatus;
25  import org.apache.shardingsphere.infra.executor.sql.process.model.ExecuteProcessUnit;
26  import org.apache.shardingsphere.infra.util.spi.ShardingSphereServiceLoader;
27  
28  import java.util.Optional;
29  
30  /**
31   * Execute process report.
32   */
33  public final class ExecuteProcessReporter {
34      
35      /**
36       * Report connect.
37       *
38       * @param executionGroupContext execution group context
39       */
40      public void reportConnect(final ExecutionGroupContext<? extends SQLExecutionUnit> executionGroupContext) {
41          ExecuteProcessContext executeProcessContext = new ExecuteProcessContext("", executionGroupContext, ExecuteProcessStatus.SLEEP);
42          ShowProcessListManager.getInstance().putProcessContext(executeProcessContext.getExecutionID(), executeProcessContext);
43      }
44      
45      /**
46       * Report execute.
47       *
48       * @param queryContext query context
49       * @param executionGroupContext execution group context
50       */
51      public void reportExecute(final QueryContext queryContext, final ExecutionGroupContext<? extends SQLExecutionUnit> executionGroupContext) {
52          ExecuteProcessContext executeProcessContext = new ExecuteProcessContext(queryContext.getSql(), executionGroupContext, ExecuteProcessStatus.START);
53          ShowProcessListManager.getInstance().putProcessContext(executeProcessContext.getExecutionID(), executeProcessContext);
54          ShowProcessListManager.getInstance().putProcessStatement(executeProcessContext.getExecutionID(), executeProcessContext.getProcessStatements());
55      }
56      
57      /**
58       * Report complete execution unit.
59       *
60       * @param executionID execution ID
61       * @param executionUnit execution unit
62       */
63      public void reportComplete(final String executionID, final SQLExecutionUnit executionUnit) {
64          ExecuteProcessUnit executeProcessUnit = new ExecuteProcessUnit(executionUnit.getExecutionUnit(), ExecuteProcessStatus.DONE);
65          ExecuteProcessContext executeProcessContext = ShowProcessListManager.getInstance().getProcessContext(executionID);
66          Optional.ofNullable(executeProcessContext.getProcessUnits().get(executeProcessUnit.getUnitID())).ifPresent(optional -> optional.setStatus(ExecuteProcessStatus.DONE));
67      }
68      
69      /**
70       * Reset report.
71       *
72       * @param executionID execution ID
73       */
74      public void reset(final String executionID) {
75          ShowProcessListManager.getInstance().removeProcessStatement(executionID);
76          ExecuteProcessContext context = ShowProcessListManager.getInstance().getProcessContext(executionID);
77          if (null == context) {
78              return;
79          }
80          for (ExecuteProcessReporterCleaner each : ShardingSphereServiceLoader.getServiceInstances(ExecuteProcessReporterCleaner.class)) {
81              each.reset(context);
82          }
83      }
84      
85      /**
86       * Remove process context.
87       *
88       * @param executionID execution ID
89       */
90      public void remove(final String executionID) {
91          ShowProcessListManager.getInstance().removeProcessStatement(executionID);
92          ShowProcessListManager.getInstance().removeProcessContext(executionID);
93      }
94  }
