74      private final CDCJobAPI jobAPI = new CDCJobAPI();
75      
76      /**
77       * Get database name by job ID.
78       *
79       * @param jobId job ID
80       * @return database
81       */
82      public String getDatabaseNameByJobId(final String jobId) {
83          return jobAPI.getJobConfiguration(jobId).getDatabaseName();
84      }
85      
86      /**
87       * Stream data.
88       *
89       * @param requestId request ID
90       * @param requestBody stream data request body
91       * @param connectionContext connection context
92       * @param channel channel
93       * @return CDC response
94       */
95      public CDCResponse streamData(final String requestId, final StreamDataRequestBody requestBody, final CDCConnectionContext connectionContext, final Channel channel) {
96          ShardingSphereDatabase database = PipelineContextManager.getProxyContext().getContextManager().getMetaDataContexts().getMetaData().getDatabase(requestBody.getDatabase());
97          ShardingSpherePreconditions.checkNotNull(database, () -> new CDCExceptionWrapper(requestId, new CDCServerException(String.format("%s database is not exists", requestBody.getDatabase()))));
98          Map<String, Set<String>> schemaTableNameMap;
99          Collection<String> tableNames;
100         Set<String> schemaTableNames = new HashSet<>();
101         if (database.getProtocolType().isSchemaAvailable()) {
102             schemaTableNameMap = CDCSchemaTableUtils.parseTableExpressionWithSchema(database, requestBody.getSourceSchemaTableList());
103             // TODO if different schema have same table names, table name may be overwritten, because the table name at sharding rule not contain schema.
104             tableNames = schemaTableNameMap.values().stream().flatMap(Collection::stream).collect(Collectors.toList());
105             schemaTableNameMap.forEach((k, v) -> v.forEach(tableName -> schemaTableNames.add(k.isEmpty() ? tableName : String.join(".", k, tableName))));
106         } else {
107             schemaTableNames.addAll(CDCSchemaTableUtils.parseTableExpressionWithoutSchema(database, requestBody.getSourceSchemaTableList().stream().map(SchemaTable::getTable)
108                     .collect(Collectors.toList())));
109             tableNames = schemaTableNames;
110         }
111         ShardingSpherePreconditions.checkState(!tableNames.isEmpty(), () -> new CDCExceptionWrapper(requestId, new NotFindStreamDataSourceTableException()));
112         Map<String, List<DataNode>> actualDataNodesMap = buildDataNodesMap(database, tableNames);
113         ShardingSpherePreconditions.checkState(!actualDataNodesMap.isEmpty(), () -> new PipelineInvalidParameterException(String.format("Not find table %s", tableNames)));
114         boolean decodeWithTx = database.getProtocolType() instanceof OpenGaussDatabaseType;
115         StreamDataParameter parameter = new StreamDataParameter(requestBody.getDatabase(), new LinkedList<>(schemaTableNames), requestBody.getFull(), actualDataNodesMap, decodeWithTx);
116         String jobId = jobAPI.createJob(parameter, CDCSinkType.SOCKET, new Properties());
117         connectionContext.setJobId(jobId);
118         startStreaming(jobId, connectionContext, channel);
119         return CDCResponseGenerator.succeedBuilder(requestId).setStreamDataResult(StreamDataResult.newBuilder().setStreamingId(jobId).build()).build();
120     }
121     
122     private Map<String, List<DataNode>> buildDataNodesMap(final ShardingSphereDatabase database, final Collection<String> tableNames) {
123         Optional<ShardingRule> shardingRule = database.getRuleMetaData().findSingleRule(ShardingRule.class);
124         Optional<SingleRule> singleRule = database.getRuleMetaData().findSingleRule(SingleRule.class);
125         ShardingSpherePreconditions.checkState(shardingRule.isPresent() || singleRule.isPresent(), () -> new NoAnyRuleExistsException(database.getName()));
126         Map<String, List<DataNode>> result = new HashMap<>();
127         // TODO support virtual data source name
128         for (String each : tableNames) {
129             if (singleRule.isPresent() && singleRule.get().getAllDataNodes().containsKey(each)) {
130                 result.put(each, new ArrayList<>(singleRule.get().getAllDataNodes().get(each)));
131                 continue;
132             }
133             shardingRule.flatMap(value -> value.findTableRule(each)).ifPresent(rule -> result.put(each, rule.getActualDataNodes()));
134         }
135         return result;
136     }
137     
138     /**
139      * Start streaming.
140      *
141      * @param jobId job ID
142      * @param channel channel
143      * @param connectionContext connection context
144      */
145     public void startStreaming(final String jobId, final CDCConnectionContext connectionContext, final Channel channel) {
146         CDCJobConfiguration cdcJobConfig = jobAPI.getJobConfiguration(jobId);
147         ShardingSpherePreconditions.checkNotNull(cdcJobConfig, () -> new PipelineJobNotFoundException(jobId));
148         if (PipelineJobCenter.isJobExisting(jobId)) {
149             PipelineJobCenter.stop(jobId);
150         }
151         ShardingSphereDatabase database = PipelineContextManager.getProxyContext().getContextManager().getMetaDataContexts().getMetaData().getDatabase(cdcJobConfig.getDatabaseName());
152         jobAPI.startJob(jobId, new CDCSocketSink(channel, database, cdcJobConfig.getSchemaTableNames()));
153         connectionContext.setJobId(jobId);
154     }
155     
156     /**
157      * Stop streaming.
158      *
159      * @param jobId job ID
160      * @param channelId channel ID
161      */
162     public void stopStreaming(final String jobId, final ChannelId channelId) {
163         if (Strings.isNullOrEmpty(jobId)) {
164             log.warn("job id is null or empty, ignored");
165             return;
166         }
167         CDCJob job = (CDCJob) PipelineJobCenter.getJob(jobId);
168         if (null == job) {
169             return;
170         }
171         if (job.getSink().identifierMatched(channelId)) {
172             log.info("close CDC job, channel id: {}", channelId);
173             PipelineJobCenter.stop(jobId);
174             jobAPI.updateJobConfigurationDisabled(jobId, true);
175         }
176     }
177     
178     /**
179      * Drop streaming.
180      *
181      * @param jobId job ID
182      */
183     public void dropStreaming(final String jobId) {
184         jobAPI.stopAndDrop(jobId);
185     }
186     
187     /**
188      * Process ack.
189      *
190      * @param requestBody request body
191      */
192     public void processAck(final AckStreamingRequestBody requestBody) {
193         CDCAckId ackId = CDCAckId.unmarshal(requestBody.getAckId());
194         CDCImporter importer = CDCImporterManager.getImporter(ackId.getImporterId());
195         if (null == importer) {
196             log.warn("Could not find importer, ack id: {}", ackId.marshal());
197             return;
198         }
199         importer.ack(ackId.marshal());
200     }
