29  import org.apache.shardingsphere.infra.yaml.config.pojo.algorithm.YamlAlgorithmConfiguration;

35  import java.util.LinkedHashMap;
36  import java.util.LinkedList;

38  /**
39   * Compatible encryptor subscriber.
40   * @deprecated compatible support will remove in next version.
41   */
42  @Deprecated
43  @SuppressWarnings("UnstableApiUsage")
44  @Setter
45  public final class CompatibleEncryptorSubscriber implements RuleChangedSubscriber {
46      
47      private ContextManager contextManager;
48      
49      /**
50       * Renew with alter encryptor.
51       *
52       * @param event alter encryptor event
53       */
54      @Subscribe
55      public synchronized void renew(final AlterCompatibleEncryptorEvent event) {
56          if (!event.getActiveVersion().equals(contextManager.getInstanceContext().getModeContextManager().getActiveVersionByKey(event.getActiveVersionKey()))) {
57              return;
58          }
59          String yamlContent = contextManager.getInstanceContext().getModeContextManager().getVersionPathByActiveVersionKey(event.getActiveVersionKey(), event.getActiveVersion());
60          AlgorithmConfiguration toBeChangedConfig = new YamlAlgorithmConfigurationSwapper().swapToObject(YamlEngine.unmarshal(yamlContent, YamlAlgorithmConfiguration.class));
61          ShardingSphereDatabase database = contextManager.getMetaDataContexts().getMetaData().getDatabases().get(event.getDatabaseName());
62          CompatibleEncryptRuleConfiguration config = database.getRuleMetaData().findSingleRule(EncryptRule.class)
63                  .map(encryptRule -> getEncryptRuleConfiguration((CompatibleEncryptRuleConfiguration) encryptRule.getConfiguration()))
64                  .orElseGet(() -> new CompatibleEncryptRuleConfiguration(new LinkedList<>(), new LinkedHashMap<>()));
65          config.getEncryptors().put(event.getItemName(), toBeChangedConfig);
66          contextManager.getInstanceContext().getEventBusContext().post(new DatabaseRuleConfigurationChangedEvent(event.getDatabaseName(), config));
67      }
68      
69      /**
70       * Renew with drop encryptor.
71       *
72       * @param event drop encryptor event
73       */
74      @Subscribe
75      public synchronized void renew(final DropCompatibleEncryptorEvent event) {
76          if (!contextManager.getMetaDataContexts().getMetaData().containsDatabase(event.getDatabaseName())) {
77              return;
78          }
79          ShardingSphereDatabase database = contextManager.getMetaDataContexts().getMetaData().getDatabases().get(event.getDatabaseName());
80          CompatibleEncryptRuleConfiguration config = (CompatibleEncryptRuleConfiguration) database.getRuleMetaData().getSingleRule(EncryptRule.class).getConfiguration();
81          config.getEncryptors().remove(event.getItemName());
82          contextManager.getInstanceContext().getEventBusContext().post(new DatabaseRuleConfigurationChangedEvent(event.getDatabaseName(), config));
83      }
84      
85      private CompatibleEncryptRuleConfiguration getEncryptRuleConfiguration(final CompatibleEncryptRuleConfiguration config) {
86          return null == config.getEncryptors() ? new CompatibleEncryptRuleConfiguration(config.getTables(), new LinkedHashMap<>()) : config;
87      }
88  }
